CREATE TABLE USERS
   (    
	COMPANYNUMBER VARCHAR2(20), 
	NAME VARCHAR2(20),
        	EMAIL VARCHAR2(400),
        	PASSWORD VARCHAR2(400),
	EMAILTIME VARCHAR2(14),
	ROLE VARCHAR2(40) DEFAULT 'USER',
        	EMAILCHECK VARCHAR2(20) DEFAULT 'Y'
   ) ;

CREATE TABLE OTHER_EVAL_DAILY
   (	
	SEQ	INT PRIMARY KEY,
	CRPRVID VARCHAR2(20), 
	RATE_DATE VARCHAR2(20), 
	CMP_CD VARCHAR2(400), 
	CMP_NM VARCHAR2(400), 
	CR_TYP VARCHAR2(20), 
	SECU_TYP VARCHAR2(40), 
	BOND_TYP VARCHAR2(20), 
	RANK VARCHAR2(100), 
	WATCH VARCHAR2(20), 
	OUTLOOK VARCHAR2(20), 
	ISSUE_NO VARCHAR2(200) DEFAULT 'N/A', 
	ISSUE_AMT VARCHAR2(20), 
	ISSUE_AMT_TYP VARCHAR2(20), 
	ISSUE_DATE VARCHAR2(20), 
	MATU_DATE VARCHAR2(20), 
	RULE_DATE VARCHAR2(20), 
	SERIES VARCHAR2(400), 
	EXPOSE_TYP VARCHAR2(100), 
	REL_CMP_NM VARCHAR2(400)
   ) ;

   COMMENT ON COLUMN OTHER_EVAL_DAILY.CRPRVID IS '평가사';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.RATE_DATE IS '등급확정일';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.CMP_CD IS '기업코드';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.CMP_NM IS '기업명';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.CR_TYP IS '평가대상';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.SECU_TYP IS '평가종류 ';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.BOND_TYP IS '사채종류';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.RANK IS '등급';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.WATCH IS 'WATCH';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.OUTLOOK IS 'OUTLOOK';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.ISSUE_NO IS '회차';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.ISSUE_AMT IS '발행액';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.ISSUE_AMT_TYP IS '발행단위';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.ISSUE_DATE IS '발행일';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.MATU_DATE IS 'MATU_DATE';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.RULE_DATE IS 'RULE_DATE';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.SERIES IS '시리즈';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.EXPOSE_TYP IS '익스포져유형';
   COMMENT ON COLUMN OTHER_EVAL_DAILY.REL_CMP_NM IS '관계기관';
